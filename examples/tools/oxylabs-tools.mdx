Enables Agno agents to scrape data using Oxylabs enterprise-grade infrastructure that allows accessing web globally without being blocked by CAPTCHAs or IP bans.

```python

from agno.agent import Agent
from agno.tools.oxylabs import OxylabsTools

# ---------------------------------------------------------------------------
# Create Agent
# ---------------------------------------------------------------------------

agent = Agent(
    tools=[OxylabsTools()],
    markdown=True,
)

# Example 1: Google Search

# ---------------------------------------------------------------------------
# Run Agent
# ---------------------------------------------------------------------------
if __name__ == "__main__":
    agent.print_response(
        "Let's search for 'latest iPhone reviews' and provide a summary of the top 3 results. ",
    )
```
